"use strict";
 
const suncalc = require("suncalc");
 
const logger = require("../util/logger");
const state = require("../state");
const getIP = require("./GetIP");
const findLocationOfIP = require("./FindLocationOfIP");
const getWifiTowers = require("./getWifiTowers");
const findLocationOfWifiTowers = require("./FindLocationOfWifiTowers");
const findLocation = require("./FindLocation")(
  getIP,
  getWifiTowers,
  findLocationOfIP,
  findLocationOfWifiTowers
);
const changeTheme = require("./changeTheme");
const scheduleUpdate = require("./scheduleUpdate");
const { NIGHT, DAY, RETRY_TIMEOUT } = require("../util/constants");
 
const main = async lastLocation => {
  const updateTimer = Date.now();
  logger.info("Updating data...");
 
  const appState = await state.getAppState();
 
  const { locationSetManually } = appState;
  let { location } = appState;
 
  if (!locationSetManually) {
    try {
      location = await findLocation();
    } catch (err) {
      logger.error(err);
    }
  }
 
  if (location) {
    let { sunrise, sunset } = suncalc.getTimes(
      new Date(),
      location.lat,
      location.lng
    );
 
    state.setAppState({
      location,
      sunrise,
      sunset
    });
 
    logger.info(`Daytime for ${location.lat},${location.lng}:`);
    logger.info(`Sunrise: ${sunrise}`);
    logger.info(`Sunset: ${sunset}`);
 
    const now = new Date();
    if (now < sunrise || now > sunset) {
      changeTheme(NIGHT);
      scheduleUpdate(sunrise);
    } else {
      changeTheme(DAY);
      scheduleUpdate(sunset);
    }
 
    logger.info(`Update took ${Date.now() - updateTimer} ms`);
  } else {
    console.log(
      'Lightswitch wasn\'t able to establish location. If you wish to set the location manually, please use "lightswitch location lat,lng" replacing "lat" and "lng" with appropriate values. Otherwise Lightswitch will retry to find your location in an hour.'
    );
    scheduleUpdate(new Date(Date.now() + RETRY_TIMEOUT));
  }
};
 
module.exports = main;
